async function handlePrivateFlow(v) { if (!privateState) { modeShown = false; return showMode(); } // STEP 1: TARGET (optional) if (privateState.step === 'TARGET') { const raw = (v || '').trim(); // EMPTY = untargeted private match if (!raw) { privateState.target = null; privateState.step = 'COLOR'; return add('HOST COLOR: [R] RED | [B] BLUE', '', true, handlePrivateFlow); } const target = raw.toUpperCase(); if (target === localStorage.getItem(KEY_USER)) { add('CANNOT TARGET SELF.', 'error-msg'); privateState = null; modeShown = false; return showMode(); } const exists = await isUsernameOccupied(target); if (!exists) { add('TARGET USER NOT FOUND.', 'error-msg'); privateState = null; modeShown = false; return showMode(); } privateState.target = target; privateState.step = 'COLOR'; return add('HOST COLOR: [R] RED | [B] BLUE', '', true, handlePrivateFlow); } // STEP 2: COLOR if (privateState.step === 'COLOR') { const input = (v || '').trim().toUpperCase(); // accept: "R", "B", "R : BLUE", etc. const colorChar = input.includes('R') ? 'R' : input.includes('B') ? 'B' : null; const color = colorChar === 'R' ? 0 : colorChar === 'B' ? 1 : null; if (color === null) { add('Unknown Command.', 'error-msg'); privateState = null; modeShown = false; return showMode(); } const target = privateState.target; privateState = null; return registerHost(color, false, { isPrivate: true, privateMode: 'HOST', privateTarget: target }); } // STEP 3: CODE JOIN if (privateState.step === 'CODE') { const code = (v || '').trim().toUpperCase(); if (!code) { add('INVALID CODE.', 'error-msg'); privateState = null; modeShown = false; return showMode(); } privateState = null; return joinPrivate(code); } privateState = null; modeShown = false; return showMode(); }
