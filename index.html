function waitForEnter(input, onSubmit) { return new Promise((resolve) => { input.addEventListener('keydown', (e) => { if (e.key !== 'Enter') return; e.preventDefault(); const value = input.value.trim().toUpperCase(); input.disabled = true; const wrapper = input.parentElement; const promptText = wrapper?.querySelector('.prompt')?.textContent ?? ''; // Echo command like a terminal const echo = appendLine('highlight'); echo.textContent = `${promptText}${value}`; // Remove input line wrapper.remove(); // Run handler asynchronously (DO NOT BLOCK UI) Promise.resolve(onSubmit(value)).catch(console.error); resolve(); }, { once: true }); }); }
